185 /// A memory-resident sparse tensor using a storage scheme based on
186 /// per-dimension sparse/dense annotations. This data structure provides
187 /// a bufferized form of a sparse tensor type. In contrast to generating
188 /// setup methods for each differently annotated sparse tensor, this
189 /// method provides a convenient "one-size-fits-all" solution that simply
190 /// takes an input tensor and annotations to implement all required setup
191 /// in a general manner.
192 template <typename P, typename I, typename V>
193 class SparseTensorStorage final : public SparseTensorStorageBase {
194  /// Private constructor to share code between the other constructors.
195  /// Beware that the object is not necessarily guaranteed to be in a
196  /// valid state after this constructor alone; e.g., `isCompressedDim(d)`
197  /// doesn't entail `!(pointers[d].empty())`.
198  ///
199  /// Precondition: `perm` and `sparsity` must be valid for `dimSizes.size()`.
200  SparseTensorStorage(const std::vector<uint64_t> &dimSizes,
201  const uint64_t *perm, const DimLevelType *sparsity)
202  : SparseTensorStorageBase(dimSizes, perm, sparsity), pointers(getRank()),
203  indices(getRank()), idx(getRank()) {}

268  /// Partially specialize lexicographical insertions based on template types.
269  void lexInsert(const uint64_t *cursor, V val) final {
270  // First, wrap up pending insertion path.
271  uint64_t diff = 0;
272  uint64_t top = 0;
273  if (!values.empty()) {
274  diff = lexDiff(cursor);
275  endPath(diff + 1);
276  top = idx[diff] + 1;
277  }
278  // Then continue with insertion path.
279  insPath(cursor, diff, top, val);
280  }
281 
282  /// Partially specialize expanded insertions based on template types.
283  /// Note that this method resets the values/filled-switch array back
284  /// to all-zero/false while only iterating over the nonzero elements.
285  void expInsert(uint64_t *cursor, V *values, bool *filled, uint64_t *added,
286  uint64_t count) final {
287  if (count == 0)
288  return;
289  // Sort.
290  std::sort(added, added + count);
291  // Restore insertion path for first insert.
292  const uint64_t lastDim = getRank() - 1;
293  uint64_t index = added[0];
294  assert(filled[index] && "added index is not filled");
295  cursor[lastDim] = index;
296  lexInsert(cursor, values[index]);
297  values[index] = 0;
298  filled[index] = false;
299  // Subsequent insertions are quick.
300  for (uint64_t i = 1; i < count; ++i) {
301  assert(index < added[i] && "non-lexicographic insertion");
302  index = added[i];
303  assert(filled[index] && "added index is not filled");
304  cursor[lastDim] = index;
305  insPath(cursor, lastDim, added[i - 1] + 1, values[index]);
306  values[index] = 0;
307  filled[index] = false;
308  }
309  }
310 
311  /// Finalizes lexicographic insertions.
312  void endInsert() final {
313  if (values.empty())
314  finalizeSegment(0);
315  else
316  endPath(0);
317  }

327  /// Returns this sparse tensor storage scheme as a new memory-resident
328  /// sparse tensor in coordinate scheme with the given dimension order.
329  ///
330  /// Precondition: `perm` must be valid for `getRank()`.
331  SparseTensorCOO<V> *toCOO(const uint64_t *perm) const {
332  SparseTensorEnumeratorBase<V> *enumerator;
333  newEnumerator(&enumerator, getRank(), perm);

335  new SparseTensorCOO<V>(enumerator->permutedSizes(), values.size());
336  enumerator->forallElements([&coo](const std::vector<uint64_t> &ind, V val) {
337  coo->add(ind, val);

340  // or if there are then that we don't filter them out.
341  // Cf., <https://github.com/llvm/llvm-project/issues/54179>

347 private:
348  /// Appends an arbitrary new position to `pointers[d]`. This method
349  /// checks that `pos` is representable in the `P` type; however, it
350  /// does not check that `pos` is semantically valid (i.e., larger than
351  /// the previous position and smaller than `indices[d].capacity()`).
352  void appendPointer(uint64_t d, uint64_t pos, uint64_t count = 1) {
353  ASSERT_COMPRESSED_DIM(d);
354  assert(pos <= std::numeric_limits<P>::max() &&
355  "Pointer value is too large for the P-type");
356  pointers[d].insert(pointers[d].end(), count, static_cast<P>(pos));
357  }
358 
359  /// Appends index `i` to dimension `d`, in the semantically general
360  /// sense. For non-dense dimensions, that means appending to the
361  /// `indices[d]` array, checking that `i` is representable in the `I`
362  /// type; however, we do not verify other semantic requirements (e.g.,
363  /// that `i` is in bounds for `dimSizes[d]`, and not previously occurring
364  /// in the same segment). For dense dimensions, this method instead
365  /// appends the appropriate number of zeros to the `values` array,
366  /// where `full` is the number of "entries" already written to `values`
367  /// for this segment (aka one after the highest index previously appended).
368  void appendIndex(uint64_t d, uint64_t full, uint64_t i) {
369  if (isCompressedDim(d) || isSingletonDim(d)) {
370  assert(i <= std::numeric_limits<I>::max() &&
371  "Index value is too large for the I-type");
372  indices[d].push_back(static_cast<I>(i));
373  } else { // Dense dimension.
374  ASSERT_DENSE_DIM(d);
375  assert(i >= full && "Index was already filled");
376  if (i == full)
377  return; // Short-circuit, since it'll be a nop.
378  if (d + 1 == getRank())
379  values.insert(values.end(), i - full, 0);
380  else
381  finalizeSegment(d + 1, 0, i - full);
382  }
383  }
384 
385  /// Writes the given coordinate to `indices[d][pos]`. This method
386  /// checks that `i` is representable in the `I` type; however, it
387  /// does not check that `i` is semantically valid (i.e., in bounds
388  /// for `dimSizes[d]` and not elsewhere occurring in the same segment).
389  void writeIndex(uint64_t d, uint64_t pos, uint64_t i) {
390  ASSERT_COMPRESSED_OR_SINGLETON_DIM(d);
391  // Subscript assignment to `std::vector` requires that the `pos`-th
392  // entry has been initialized; thus we must be sure to check `size()`
393  // here, instead of `capacity()` as would be ideal.
394  assert(pos < indices[d].size() && "Index position is out of bounds");
395  assert(i <= std::numeric_limits<I>::max() &&
396  "Index value is too large for the I-type");
397  indices[d][pos] = static_cast<I>(i);
398  }

400  /// Computes the assembled-size associated with the `d`-th dimension,
401  /// given the assembled-size associated with the `(d-1)`-th dimension.
402  /// "Assembled-sizes" correspond to the (nominal) sizes of overhead
403  /// storage, as opposed to "dimension-sizes" which are the cardinality
404  /// of coordinates for that dimension.
405  ///
406  /// Precondition: the `pointers[d]` array must be fully initialized
407  /// before calling this method.
408  uint64_t assembledSize(uint64_t parentSz, uint64_t d) const {
409  if (isCompressedDim(d))
410  return pointers[d][parentSz];
411  if (isSingletonDim(d))
412  return parentSz; // New size is same as the parent.
413  if (isDenseDim(d))
414  return parentSz * getDimSizes()[d];
415  MLIR_SPARSETENSOR_FATAL("unsupported dimension level type");
416  }

418  /// Initializes sparse tensor storage scheme from a memory-resident sparse
419  /// tensor in coordinate scheme. This method prepares the pointers and
420  /// indices arrays under the given per-dimension dense/sparse annotations.
421  ///
422  /// Preconditions:
423  /// (1) the `elements` must be lexicographically sorted.
424  /// (2) the indices of every element are valid for `dimSizes` (equal rank
425  /// and pointwise less-than).
426  void fromCOO(const std::vector<Element<V>> &elements, uint64_t lo,
427  uint64_t hi, uint64_t d) {
428  const uint64_t rank = getRank();
429  assert(d <= rank && hi <= elements.size());
430  // Once dimensions are exhausted, insert the numerical values.
431  if (d == rank) {
432  assert(lo < hi);
433  values.push_back(elements[lo].value);
434  return;
435  }
436  // Visit all elements in this interval.
437  uint64_t full = 0;
438  while (lo < hi) { // If `hi` is unchanged, then `lo < elements.size()`.
439  // Find segment in interval with same index elements in this dimension.
440  const uint64_t i = elements[lo].indices[d];
441  uint64_t seg = lo + 1;
442  if (isUniqueDim(d))
443  while (seg < hi && elements[seg].indices[d] == i)
444  ++seg;
445  // Handle segment in interval for sparse or dense dimension.
446  appendIndex(d, full, i);
447  full = i + 1;
448  fromCOO(elements, lo, seg, d + 1);
449  // And move on to next segment in interval.
450  lo = seg;
451  }
452  // Finalize the sparse pointer structure at this dimension.
453  finalizeSegment(d, full);
454  }
455 
456  /// Finalizes the sparse pointer structure at this dimension.
457  void finalizeSegment(uint64_t d, uint64_t full = 0, uint64_t count = 1) {
458  if (count == 0)
459  return; // Short-circuit, since it'll be a nop.
460  if (isCompressedDim(d)) {
461  appendPointer(d, indices[d].size(), count);
462  } else if (isSingletonDim(d)) {
463  return; // Nothing to finalize.
464  } else { // Dense dimension.
465  ASSERT_DENSE_DIM(d);
466  const uint64_t sz = getDimSizes()[d];
467  assert(sz >= full && "Segment is overfull");
468  count = detail::checkedMul(count, sz - full);
469  // For dense storage we must enumerate all the remaining coordinates
470  // in this dimension (i.e., coordinates after the last non-zero
471  // element), and either fill in their zero values or else recurse
472  // to finalize some deeper dimension.
473  if (d + 1 == getRank())
474  values.insert(values.end(), count, 0);
475  else
476  finalizeSegment(d + 1, 0, count);
477  }
478  }
479 
480  /// Wraps up a single insertion path, inner to outer.
481  void endPath(uint64_t diff) {
482  const uint64_t rank = getRank();
483  assert(diff <= rank && "Dimension-diff is out of bounds");
484  for (uint64_t i = 0; i < rank - diff; ++i) {
485  const uint64_t d = rank - i - 1;
486  finalizeSegment(d, idx[d] + 1);
487  }
488  }
489 
490  /// Continues a single insertion path, outer to inner.
491  void insPath(const uint64_t *cursor, uint64_t diff, uint64_t top, V val) {
492  const uint64_t rank = getRank();
493  assert(diff <= rank && "Dimension-diff is out of bounds");
494  for (uint64_t d = diff; d < rank; ++d) {
495  const uint64_t i = cursor[d];
496  appendIndex(d, top, i);
497  top = 0;
498  idx[d] = i;
499  }
500  values.push_back(val);
501  }
502 
503  /// Finds the lexicographic differing dimension.
504  uint64_t lexDiff(const uint64_t *cursor) const {
505  const uint64_t rank = getRank();
506  for (uint64_t r = 0; r < rank; ++r)
507  if (cursor[r] > idx[r])
508  return r;
509  else
510  assert(cursor[r] == idx[r] && "non-lexicographic insertion");
511  assert(0 && "duplication insertion");
512  return -1u;
513  }

531 /// A (higher-order) function object for enumerating the elements of some
532 /// `SparseTensorStorage` under a permutation. That is, the `forallElements`
533 /// method encapsulates the loop-nest for enumerating the elements of
534 /// the source tensor (in whatever order is best for the source tensor),
535 /// and applies a permutation to the coordinates/indices before handing
536 /// each element to the callback. A single enumerator object can be
537 /// freely reused for several calls to `forallElements`, just so long
538 /// as each call is sequential with respect to one another.
539 ///
540 /// N.B., this class stores a reference to the `SparseTensorStorageBase`
541 /// passed to the constructor; thus, objects of this class must not
542 /// outlive the sparse tensor they depend on.
543 ///
544 /// Design Note: The reason we define this class instead of simply using
545 /// `SparseTensorEnumerator<P,I,V>` is because we need to hide/generalize
546 /// the `<P,I>` template parameters from MLIR client code (to simplify the
547 /// type parameters used for direct sparse-to-sparse conversion). And the
548 /// reason we define the `SparseTensorEnumerator<P,I,V>` subclasses rather
549 /// than simply using this class, is to avoid the cost of virtual-method
550 /// dispatch within the loop-nest.
551 template <typename V>
552 class SparseTensorEnumeratorBase {
553 public:
554  /// Constructs an enumerator with the given permutation for mapping
555  /// the semantic-ordering of dimensions to the desired target-ordering.
556  ///
557  /// Preconditions:
558  /// * the `tensor` must have the same `V` value type.
559  /// * `perm` must be valid for `rank`.
560  SparseTensorEnumeratorBase(const SparseTensorStorageBase &tensor,
561  uint64_t rank, const uint64_t *perm)
562  : src(tensor), permsz(src.getRev().size()), reord(getRank()),
563  cursor(getRank()) {
564  assert(perm && "Received nullptr for permutation");
565  assert(rank == getRank() && "Permutation rank mismatch");
566  const auto &rev = src.getRev(); // source-order -> semantic-order
567  const auto &dimSizes = src.getDimSizes(); // in source storage-order
568  for (uint64_t s = 0; s < rank; ++s) { // `s` source storage-order
569  uint64_t t = perm[rev[s]]; // `t` target-order
570  reord[s] = t;
571  permsz[t] = dimSizes[s];
572  }
573  }
574 
575  virtual ~SparseTensorEnumeratorBase() = default;
576 
577  // We disallow copying to help avoid leaking the `src` reference.
578  // (In addition to avoiding the problem of slicing.)
579  SparseTensorEnumeratorBase(const SparseTensorEnumeratorBase &) = delete;
580  SparseTensorEnumeratorBase &
581  operator=(const SparseTensorEnumeratorBase &) = delete;
582 
583  /// Returns the source/target tensor's rank. (The source-rank and
584  /// target-rank are always equal since we only support permutations.
585  /// Though once we add support for other dimension mappings, this
586  /// method will have to be split in two.)
587  uint64_t getRank() const { return permsz.size(); }
588 
589  /// Returns the target tensor's dimension sizes.
590  const std::vector<uint64_t> &permutedSizes() const { return permsz; }
591 
592  /// Enumerates all elements of the source tensor, permutes their
593  /// indices, and passes the permuted element to the callback.
594  /// The callback must not store the cursor reference directly,
595  /// since this function reuses the storage. Instead, the callback
596  /// must copy it if they want to keep it.
597  virtual void forallElements(ElementConsumer<V> yield) = 0;
598 
599 protected:
600  const SparseTensorStorageBase &src;
601  std::vector<uint64_t> permsz; // in target order.
602  std::vector<uint64_t> reord; // source storage-order -> target order.
603  std::vector<uint64_t> cursor; // in target order.
604 };

607 template <typename P, typename I, typename V>
608 class SparseTensorEnumerator final : public SparseTensorEnumeratorBase<V> {
609  using Base = SparseTensorEnumeratorBase<V>;
610  using StorageImpl = SparseTensorStorage<P, I, V>;
611 
612 public:
613  /// Constructs an enumerator with the given permutation for mapping
614  /// the semantic-ordering of dimensions to the desired target-ordering.
615  ///
616  /// Precondition: `perm` must be valid for `rank`.
617  SparseTensorEnumerator(const StorageImpl &tensor, uint64_t rank,
618  const uint64_t *perm)
619  : Base(tensor, rank, perm) {}
620 
621  ~SparseTensorEnumerator() final = default;
622 
623  void forallElements(ElementConsumer<V> yield) final {
624  forallElements(yield, 0, 0);
625  }
626 
627 private:
628  /// The recursive component of the public `forallElements`.
629  void forallElements(ElementConsumer<V> yield, uint64_t parentPos,
630  uint64_t d) {
631  // Recover the `<P,I,V>` type parameters of `src`.
632  const auto &src = static_cast<const StorageImpl &>(this->src);
633  if (d == Base::getRank()) {
634  assert(parentPos < src.values.size() &&
635  "Value position is out of bounds");
636  // TODO: <https://github.com/llvm/llvm-project/issues/54179>
637  yield(this->cursor, src.values[parentPos]);
638  return;
639  }
640  const auto dlt = src.getDimType(d);
641  if (isCompressedDLT(dlt)) {
642  // Look up the bounds of the `d`-level segment determined by the
643  // `d-1`-level position `parentPos`.
644  const std::vector<P> &pointersD = src.pointers[d];
645  assert(parentPos + 1 < pointersD.size() &&
646  "Parent pointer position is out of bounds");
647  const uint64_t pstart = static_cast<uint64_t>(pointersD[parentPos]);
648  const uint64_t pstop = static_cast<uint64_t>(pointersD[parentPos + 1]);
649  // Loop-invariant code for looking up the `d`-level coordinates/indices.
650  const std::vector<I> &indicesD = src.indices[d];
651  assert(pstop <= indicesD.size() && "Index position is out of bounds");
652  uint64_t &cursorReordD = this->cursor[this->reord[d]];
653  for (uint64_t pos = pstart; pos < pstop; ++pos) {
654  cursorReordD = static_cast<uint64_t>(indicesD[pos]);
655  forallElements(yield, pos, d + 1);
656  }
657  } else if (isSingletonDLT(dlt)) {
658  this->cursor[this->reord[d]] = src.getIndex(d, parentPos);
659  forallElements(yield, parentPos, d + 1);
660  } else {
661  assert(isDenseDLT(dlt)); // TODO: reuse the ASSERT_DENSE_DIM message
662  const uint64_t sz = src.getDimSizes()[d];
663  const uint64_t pstart = parentPos * sz;
664  uint64_t &cursorReordD = this->cursor[this->reord[d]];
665  for (uint64_t i = 0; i < sz; ++i) {
666  cursorReordD = i;
667  forallElements(yield, pstart + i, d + 1);
668  }
669  }
670  }
671 };

674 /// Statistics regarding the number of nonzero subtensors in
675 /// a source tensor, for direct sparse=>sparse conversion a la
676 /// <https://arxiv.org/abs/2001.02609>.
677 ///
678 /// N.B., this class stores references to the parameters passed to
679 /// the constructor; thus, objects of this class must not outlive
680 /// those parameters.
681 class SparseTensorNNZ final {
682 public:
683  /// Allocates the statistics structure for the desired sizes and
684  /// sparsity (in the target tensor's storage-order). This constructor
685  /// does not actually populate the statistics, however; for that see
686  /// `initialize`.
687  ///
688  /// Precondition: `dimSizes` must not contain zeros.
689  SparseTensorNNZ(const std::vector<uint64_t> &dimSizes,
690  const std::vector<DimLevelType> &sparsity);
691 
692  // We disallow copying to help avoid leaking the stored references.
693  SparseTensorNNZ(const SparseTensorNNZ &) = delete;
694  SparseTensorNNZ &operator=(const SparseTensorNNZ &) = delete;
695 
696  /// Returns the rank of the target tensor.
697  uint64_t getRank() const { return dimSizes.size(); }
698 
699  /// Enumerates the source tensor to fill in the statistics. The
700  /// enumerator should already incorporate the permutation (from
701  /// semantic-order to the target storage-order).
702  template <typename V>
703  void initialize(SparseTensorEnumeratorBase<V> &enumerator) {
704  assert(enumerator.getRank() == getRank() && "Tensor rank mismatch");
705  assert(enumerator.permutedSizes() == dimSizes && "Tensor size mismatch");
706  enumerator.forallElements(
707  [this](const std::vector<uint64_t> &ind, V) { add(ind); });
708  }
709 
710  /// The type of callback functions which receive an nnz-statistic.
711  using NNZConsumer = const std::function<void(uint64_t)> &;
712 
713  /// Lexicographically enumerates all indicies for dimensions strictly
714  /// less than `stopDim`, and passes their nnz statistic to the callback.
715  /// Since our use-case only requires the statistic not the coordinates
716  /// themselves, we do not bother to construct those coordinates.
717  void forallIndices(uint64_t stopDim, NNZConsumer yield) const;
718 
719 private:
720  /// Adds a new element (i.e., increment its statistics). We use
721  /// a method rather than inlining into the lambda in `initialize`,
722  /// to avoid spurious templating over `V`. And this method is private
723  /// to avoid needing to re-assert validity of `ind` (which is guaranteed
724  /// by `forallElements`).
725  void add(const std::vector<uint64_t> &ind);
726 
727  /// Recursive component of the public `forallIndices`.
728  void forallIndices(NNZConsumer yield, uint64_t stopDim, uint64_t parentPos,
729  uint64_t d) const;
730 
731  // All of these are in the target storage-order.
732  const std::vector<uint64_t> &dimSizes;
733  const std::vector<DimLevelType> &dimTypes;
734  std::vector<std::vector<uint64_t>> nnz;
735 };

788 template <typename P, typename I, typename V>
789 SparseTensorStorage<P, I, V>::SparseTensorStorage(
790  const std::vector<uint64_t> &dimSizes, const uint64_t *perm,
791  const DimLevelType *sparsity, SparseTensorCOO<V> *coo)
792  : SparseTensorStorage(dimSizes, perm, sparsity) {
793  // Provide hints on capacity of pointers and indices.
794  // TODO: needs much fine-tuning based on actual sparsity; currently
795  // we reserve pointer/index space based on all previous dense
796  // dimensions, which works well up to first sparse dim; but
797  // we should really use nnz and dense/sparse distribution.
798  bool allDense = true;
799  uint64_t sz = 1;
800  for (uint64_t r = 0, rank = getRank(); r < rank; ++r) {
801  if (isCompressedDim(r)) {
802  // TODO: Take a parameter between 1 and `dimSizes[r]`, and multiply
803  // `sz` by that before reserving. (For now we just use 1.)
804  pointers[r].reserve(sz + 1);
805  pointers[r].push_back(0);
806  indices[r].reserve(sz);
807  sz = 1;
808  allDense = false;
809  } else if (isSingletonDim(r)) {
810  indices[r].reserve(sz);
811  sz = 1;
812  allDense = false;
813  } else { // Dense dimension.
814  ASSERT_DENSE_DIM(r);
815  sz = detail::checkedMul(sz, getDimSizes()[r]);
816  }
817  }
818  // Then assign contents from coordinate scheme tensor if provided.
819  if (coo) {
820  // Ensure both preconditions of `fromCOO`.
821  assert(coo->getDimSizes() == getDimSizes() && "Tensor size mismatch");
822  coo->sort();
823  // Now actually insert the `elements`.
824  const std::vector<Element<V>> &elements = coo->getElements();
825  uint64_t nnz = elements.size();
826  values.reserve(nnz);
827  fromCOO(elements, 0, nnz, 0);
828  } else if (allDense) {
829  values.resize(sz, 0);
830  }
831 }

833 template <typename P, typename I, typename V>
834 SparseTensorStorage<P, I, V>::SparseTensorStorage(
835  const std::vector<uint64_t> &dimSizes, const uint64_t *perm,
836  const DimLevelType *sparsity, const SparseTensorStorageBase &tensor)
837  : SparseTensorStorage(dimSizes, perm, sparsity) {
838  SparseTensorEnumeratorBase<V> *enumerator;
839  tensor.newEnumerator(&enumerator, getRank(), perm);
840  {
841  // Initialize the statistics structure.
842  SparseTensorNNZ nnz(getDimSizes(), getDimTypes());
843  nnz.initialize(*enumerator);
844  // Initialize "pointers" overhead (and allocate "indices", "values").
845  uint64_t parentSz = 1; // assembled-size (not dimension-size) of `r-1`.
846  for (uint64_t rank = getRank(), r = 0; r < rank; ++r) {
847  if (isCompressedDim(r)) {
848  pointers[r].reserve(parentSz + 1);
849  pointers[r].push_back(0);
850  uint64_t currentPos = 0;
851  nnz.forallIndices(r, [this, &currentPos, r](uint64_t n) {
852  currentPos += n;
853  appendPointer(r, currentPos);
854  });
855  assert(pointers[r].size() == parentSz + 1 &&
856  "Final pointers size doesn't match allocated size");
857  // That assertion entails `assembledSize(parentSz, r)`
858  // is now in a valid state. That is, `pointers[r][parentSz]`
859  // equals the present value of `currentPos`, which is the
860  // correct assembled-size for `indices[r]`.
861  }
862  // Update assembled-size for the next iteration.
863  parentSz = assembledSize(parentSz, r);
864  // Ideally we need only `indices[r].reserve(parentSz)`, however
865  // the `std::vector` implementation forces us to initialize it too.
866  // That is, in the yieldPos loop we need random-access assignment
867  // to `indices[r]`; however, `std::vector`'s subscript-assignment
868  // only allows assigning to already-initialized positions.
869  if (isCompressedDim(r) || isSingletonDim(r))
870  indices[r].resize(parentSz, 0);
871  else
872  ASSERT_DENSE_DIM(r); // Future-proofing.
873  }
874  values.resize(parentSz, 0); // Both allocate and zero-initialize.
875  }
876  // The yieldPos loop
877  enumerator->forallElements([this](const std::vector<uint64_t> &ind, V val) {
878  uint64_t parentSz = 1, parentPos = 0;
879  for (uint64_t rank = getRank(), r = 0; r < rank; ++r) {
880  if (isCompressedDim(r)) {
881  // If `parentPos == parentSz` then it's valid as an array-lookup;
882  // however, it's semantically invalid here since that entry
883  // does not represent a segment of `indices[r]`. Moreover, that
884  // entry must be immutable for `assembledSize` to remain valid.
885  assert(parentPos < parentSz && "Pointers position is out of bounds");
886  const uint64_t currentPos = pointers[r][parentPos];
887  // This increment won't overflow the `P` type, since it can't
888  // exceed the original value of `pointers[r][parentPos+1]`
889  // which was already verified to be within bounds for `P`
890  // when it was written to the array.
891  pointers[r][parentPos]++;
892  writeIndex(r, currentPos, ind[r]);
893  parentPos = currentPos;
894  } else if (isSingletonDim(r)) {
895  writeIndex(r, parentPos, ind[r]);
896  // the new parentPos equals the old parentPos.
897  } else { // Dense dimension.
898  ASSERT_DENSE_DIM(r);
899  parentPos = parentPos * getDimSizes()[r] + ind[r];
900  }
901  parentSz = assembledSize(parentSz, r);
902  }
903  assert(parentPos < values.size() && "Value position is out of bounds");
904  values[parentPos] = val;
905  });
906  // No longer need the enumerator, so we'll delete it ASAP.
907  delete enumerator;
908  // The finalizeYieldPos loop
909  for (uint64_t parentSz = 1, rank = getRank(), r = 0; r < rank; ++r) {
910  if (isCompressedDim(r)) {
911  assert(parentSz == pointers[r].size() - 1 &&
912  "Actual pointers size doesn't match the expected size");
913  // Can't check all of them, but at least we can check the last one.
914  assert(pointers[r][parentSz - 1] == pointers[r][parentSz] &&
915  "Pointers got corrupted");
916  // TODO: optimize this by using `memmove` or similar.
917  for (uint64_t n = 0; n < parentSz; ++n) {
918  const uint64_t parentPos = parentSz - n;
919  pointers[r][parentPos] = pointers[r][parentPos - 1];
920  }
921  pointers[r][0] = 0;
922  } else {
923  // Both dense and singleton are no-ops for the finalizeYieldPos loop.
924  // This assertion is for future-proofing.
925  assert((isDenseDim(r) || isSingletonDim(r)) &&
926  "Dimension is neither dense nor singleton");
927  }
928  parentSz = assembledSize(parentSz, r);
929  }
930 }
