30 SparseTensorStorageBase::SparseTensorStorageBase(
31  const std::vector<uint64_t> &dimSizes, const uint64_t *perm,
32  const DimLevelType *sparsity)
33  : dimSizes(dimSizes), rev(getRank()),
34  dimTypes(sparsity, sparsity + getRank()) {
35  assert(perm && "Got nullptr for permutation");
36  assert(sparsity && "Got nullptr for sparsity");
37  const uint64_t rank = getRank();
38  // Validate parameters.
39  assert(rank > 0 && "Trivial shape is unsupported");
40  for (uint64_t r = 0; r < rank; ++r) {
41  assert(dimSizes[r] > 0 && "Dimension size zero has trivial storage");
42  assert((isDenseDim(r) || isCompressedDim(r) || isSingletonDim(r)) &&
43  "Unsupported DimLevelType");
44  }
45  // Construct the "reverse" (i.e., inverse) permutation.
46  // TODO: should move this computation off to the codegen
47  for (uint64_t r = 0; r < rank; ++r)
48  rev[perm[r]] = r;
49 }

108 void mlir::sparse_tensor::detail::assertPermutedSizesMatchShape(
